
CREATE TABLE Acorde_Nota(
	AcordeId INT,
    NotaId INT,
    PRIMARY KEY(AcordeId, NotaId),
    FOREIGN KEY (AcordeId) REFERENCES Acorde(Id),
    FOREIGN KEY (NotaId) REFERENCES  Notas(Id)
);

CREATE TABLE Escala(
    EscalaId INT AUTO_INCREMENT PRIMARY KEY,
    Intervalos VARCHAR(30),
    Notas VARCHAR(30) NOT NULL,
    FOREIGN KEY (EscalaId) REFERENCES Escala(EscalaId)
);


CREATE TABLE CampoHarmonico_Acorde(
	AcordeId INT,
    CampoHarmonicoId INT,
    PRIMARY KEY(AcordeId, CampoHarmonicoId),
    FOREIGN KEY(AcordeId) REFERENCES Acorde(AcordeId),
    FOREIGN KEY (CampoHarmonicoId) REFERENCES CampoHarmonico(CampoHarmonicoId)
);
